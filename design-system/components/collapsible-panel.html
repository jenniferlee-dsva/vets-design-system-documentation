Collapsible Panel

AdditionalInfo components are interactive components that reveal and hide content from the page.

import React from 'react';
import CollapsiblePanel from './CollapsiblePanel.jsx';

export default function CollapsiblePanelExample(props) {
  return (
    <div>
      <CollapsiblePanel
        panelName={props.panelName}>
        <div>This panel defaults to closed.</div>
      </CollapsiblePanel>

      <CollapsiblePanel
        panelName={props.panelName} startOpen>
        <div>This panel defaults to open.</div>
      </CollapsiblePanel>
    </div>
  );
}
